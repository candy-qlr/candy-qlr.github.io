# 2. 手写 call

  • 第一个参数是函数的上下文 this 指向的对象
  • 剩余参数
Function.prototype.mycall = function(context, ...args){
    context = context ?? window // undefined或者null返回window
    context._fn = this
    const res = context._fn(...args)
    delete context._fn
    return res
}
